Mirroring the operator bundle to internal quay
  • First, create the ImageSetConfiguration 
kind: ImageSetConfiguration
apiVersion: mirror.openshift.io/v1alpha2
mirror:
  operators:
    - catalog: registry.redhat.io/redhat/certified-operator-index:v4.21
      packages:
        - name: portworx-certified # This name come from oc get sub, it will be under the package column
          channels:
            - name: stable # Or the specific channel you require
  • Mirror the images
[root@mirror imageset]# oc-mirror -c imageset.yaml \
  docker://mirror.ocp4.plus.local:8443 \
  --v2 \
  --workspace file:///oc-mirror-workspace #The files will be created here\
  --dest-tls-verify=false \
  --authfile config.json # you can get this file by using podman login and copy the auth file \
  --remove-signatures \
  --retry-times 10 \
  --log-level debug
  • In the disconnected cluster add the custom portworx catalog, these files will be created automatically by oc-mirror command in the worspace
# 1. Update the image pull redirects (triggers node rollouts)
oc apply -f /oc-mirror-workspace/working-dir/cluster-resources/idms-oc-mirror.yaml

# 2. Add the Portworx catalog to OperatorHub
oc apply -f /oc-mirror-workspace/working-dir/cluster-resources/cs-certified-operator-index-v4-21.yaml
  • If not set, disable the default catalogs
oc patch operatorhub cluster --type='json' -p='[{"op": "add", "path": "/spec/disableAllDefaultSources", "value": true}]'
