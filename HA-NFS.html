Add the HA rhel repository on both nodes
subscription-manager repos --list | grep -i rhel-9-for-x86_64-highavailability-rpms
subscription-manager repos --enable=rhel-9-for-x86_64-highavailability-rpms
Install Cluster Packages on both nodes
# Install Pacemaker, Corosync, and NFS packages

sudo dnf install -y pcs pacemaker corosync fence-agents-all nfs-utils

# Enable and start the cluster management service
sudo systemctl enable --now pcsd
sudo systemctl disable --now nfs-server
systemctl enable --now pacemaker
systemctl enable --now corosync
Set Cluster Authentication
# Set the hacluster user password on both nodes
sudo passwd hacluster

# Authenticate the nodes (run on one node)
sudo pcs host auth 10.100.100.200 10.100.100.202 -u hacluster

partition the disk or use separate disks for the SBD and data

  • Install the SBD 
sudo dnf install sbd #on both nodes
  • Initialize the SBD disk
sbd -d /dev/sdc1 create # on one node

modprobe softdog

  • Verify the device (Run on both nodes):

Create the Cluster on one node
# Create and start the cluster
sudo pcs cluster setup ha-nfs 10.100.100.200 10.100.100.202

# Start the cluster on all nodes
sudo pcs cluster start --all

# Enable cluster to start on boot
sudo pcs cluster enable --all

# Verify cluster status
sudo pcs status
Prepare Shared Storage 

# On both  nodes

# Configuration option global/system_id_source.
system_id_source = "uname"

# verify on both nodes

# lvm systemid
  system ID: z1.example.com
# uname -n
  z1.example.com
  • On one node
pvcreate /dev/sdb1
vgcreate --setautoactivation n my_vg /dev/sdb1
  • Verify that the new volume group has the system ID of the node on which you are running and from which you created the volume group.
vgs -o+systemid
  VG    #PV #LV #SN Attr   VSize  VFree  System ID
  my_vg   1   0   0 wz--n- <1.82t <1.82t z1.example.com
  • Create a logical volume using the volume group my_vg on onde node
lvcreate -L450 -n my_lv my_vg

  Logical volume "my_lv" created
  • Create an XFS file system on the logical volume my_lv.
mkfs.xfs /dev/my_vg/my_lv
  • On the other node
lvmdevices --adddev /dev/sdb1
  • on both nodes edit the lvm.conf to add all vg expect the one you created
auto_activation_volume_list = [ "rhel_root", "rhel_home" ]
  • Rebuild the initramfs boot image to guarantee that the boot image will not try to activate a volume group controlled by the cluster.  on both nodes
dracut -H -f /boot/initramfs-$(uname -r).img $(uname -r)
reboot # reboot the nodes
Create Cluster Resources on one node
# Create the filesystem resource
sudo pcs resource create nfs_fs ocf:heartbeat:Filesystem \
    device="/dev/my_vg/my_lv" directory="/srv/nfs/shared" fstype="xfs" \
    --group nfs_group

# Create the virtual IP resource
sudo pcs resource create nfs_vip ocf:heartbeat:IPaddr2 \
    ip=10.100.100.203 cidr_netmask=24 \
    --group nfs_group

# Create the NFS server resource
sudo pcs resource create nfs_server ocf:heartbeat:nfsserver \
    nfs_shared_infodir=/srv/nfs/nfsinfo \
    --group nfs_group

# Create the NFS export resource
sudo pcs resource create nfs_export ocf:heartbeat:exportfs \
    clientspec="10.100.100.0/24" options="rw,sync,no_root_squash" \
    directory="/srv/nfs/shared" fsid=1 \
    --group nfs_group
sudo pcs resource create nfs_lvm ocf:heartbeat:LVM-activate \
    vgname="nfs_vg" \
    vg_access_mode="passthrough" \
    op monitor interval=30s
