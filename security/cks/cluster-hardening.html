Certificates APIs

  • CSR request to sign certificates
apiVersion: certificates.k8s.io/v1
kind: CertificateSigningRequest
metadata:
  name: user1-csr
spec:
  request: $(cat user1.csr | base64 | tr -d '\n') >> the user creates his  own key and csr files
  signerName: kubernetes.io/kube-apiserver-client
  usages:
  - client auth

 

kubectl get csr
kubectl certificate approve <name>
kubectl get csr <name> -o yaml >> the signed certificate is in the status 64 base coded

 

  • The controller manager is the responsible component for signing the requests so the CA key and cert can be found in the kube-controller-manager manifest file
    - --cluster-signing-cert-file=/etc/kubernetes/pki/ca.crt
    - --cluster-signing-key-file=/etc/kubernetes/pki/ca.key

Tokens for SA
  • To create a token for a SA
apiVersion: v1
kind: ServiceAccount
metadata:
  name: john
  namespace: default
secrets: 
  - name: john-secret
---
apiVersion: v1
kind: Secret
metadata:
  name: john-secret
  namespace: default
  annotations:
    kubernetes.io/service-account.name: john
type: kubernetes.io/service-account-token

 

  • Retrieve the service account token:
# Get the secret name
kubectl get serviceaccount john -o jsonpath='{.secrets[0].name}'

# Retrieve the token
kubectl get secret john-secret -o jsonpath='{.data.token}' | base64 --decode

# Save the token to a file
kubectl get secret john-secret -o jsonpath='{.data.token}' | base64 --decode > john-secret.txt

 

export SA_TOKEN=$(cat john-secret.txt)
kubectl config set-credentials john --token=$SA_TOKEN
